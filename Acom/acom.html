        def binary_search(A, index, start, end):
            mid = int((start + end) / 2)
            if index == A[mid]:
                return mid
            elif start == end + 1:
                if A[mid] < index:
                    return mid
                else:
                    return -1
            elif index < A[mid]:
                return binary_search(A, index, 0, mid - 1)
            else:
                return binary_search(A, index, mid + 1, end)
