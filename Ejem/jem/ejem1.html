          def fibonacci(n):
            fib = []
            a, b = 0, 1
            for i in range(n):
              fib.append(a)
              a, b = b, a + b
            return fib
          n = int(input("Ingrese la cantidad de números de la serie de Fibonacci que desea mostrar: "))
          result = fibonacci(n)
          print("Serie de Fibonacci con", n ,"términos:")
          print(result)
