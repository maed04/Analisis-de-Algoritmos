Carácter no repetido

        from collections import Counter

        def non(s):
            d=Counter(s)
            print(d)
            for i,j in enumerate(s):
                print(d[j])
                if d[j]==1:
                    return i
            return -1

        result = non("a ee iii oooo u")
        print("El índice del primer carácter no repetido es:", result)
