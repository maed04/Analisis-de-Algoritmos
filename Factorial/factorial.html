Factorial Iterativo

        def factorial_iterativo(n):
          fact = 1
          for i in range(1, n + 1):
            fact = fact * i
          return fact


        if __name__ == '__main__':

          n = int(input("Dame un numero: "))
          print(f'The Factorial of {n} is {factorial_iterativo(n)}')

Factorial Recursivo

        def factorial_recursivo(n):
          if n == 0:
            return 1
          else:
            return n * factorial_recursivo(n - 1)

        if __name__ == '__main__':
