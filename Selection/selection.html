Selection sort

        arr = [4, 2, 7, 12, 9, 1, 16000, -79, 1564, -950, 164, 16, 0, -526]

        for i in range(0, len(arr)):
          minimo = i
          for j in range(i+1, len(arr)):
            if arr[minimo] > arr[j]:
              minimo = j
          temp = arr[i]
          arr[i] = arr[minimo]
          arr[minimo] = temp

        print(arr)
