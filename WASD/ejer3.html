        frase = "Carlos es mi amigo"
        def quitar_vocales(frase):
          vocales = "aeiouAEIOU"
          frase_sin_vocales = ""
          for letra in frase:
            if letra not in vocales:
              frase_sin_vocales += letra

          return frase_sin_vocales

        resultado = quitar_vocales(frase)
        print("Frase sin vocales:", resultado)
